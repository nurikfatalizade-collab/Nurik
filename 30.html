Python
import json
import logging

logging.basicConfig(
    filename="shelter.log",
    level=logging.INFO,
    format="%(asctime)s - %(message)s"
)

class Animal:
    def __init__(self, name, age, weight):
        self.name = name
        self.age = age
        self.weight = weight
        self.__health = 100

    def speak(self):
        return "..."

    def eat(self, food):
        self.weight += 1
        logging.info(f"{self.name} поел {food}")

    def info(self):
        return f"{self.name}, {self.age} лет, {self.weight} кг"

    def to_dict(self):
        return {
            "type": self.__class__.__name__,
            "name": self.name,
            "age": self.age,
            "weight": self.weight
        }

    def __str__(self):
        return self.info()


class Dog(Animal):
    def speak(self):
        return "Гав!"


class Cat(Animal):
    def speak(self):
        return "Мяу!"


class Parrot(Animal):
    def speak(self):
        return "Привет!"


class Shelter:
    def __init__(self):
        self.animals = []

    def add_animal(self, animal):
        self.animals.append(animal)
        logging.info(f"Добавлено животное {animal.name}")

    def remove_animal(self, name):
        for a in self.animals:
            if a.name == name:
                self.animals.remove(a)
                logging.info(f"Удалено животное {name}")
                return
        print("Животное не найдено")

    def show_all(self):
        if not self.animals:
            print("Животных нет")
        for a in self.animals:
            print(a, "-", a.speak())

    def save_to_file(self):
        data = [a.to_dict() for a in self.animals]
        with open("shelter.json", "w") as f:
            json.dump(data, f)

    def load_from_file(self):
        try:
            with open("shelter.json", "r") as f:
                data = json.load(f)

            for item in data:
                if item["type"] == "Dog":
                    animal = Dog(item["name"], item["age"], item["weight"])
                elif item["type"] == "Cat":
                    animal = Cat(item["name"], item["age"], item["weight"])
                else:
                    animal = Parrot(item["name"], item["age"], item["weight"])

                self.animals.append(animal)

        except:
            print("Файл не найден")


def main():
    shelter = Shelter()
    shelter.load_from_file()

    while True:
        print("\n1. Добавить животное")
        print("2. Удалить животное")
        print("3. Показать всех")
        print("4. Сохранить")
        print("5. Выход")

        choice = input("Выбор: ")

        if choice == "1":
            t = input("Тип (dog/cat/parrot): ")
            name = input("Имя: ")
            age = int(input("Возраст: "))
            weight = float(input("Вес: "))

            if t == "dog":
                animal = Dog(name, age, weight)
            elif t == "cat":
                animal = Cat(name, age, weight)
            else:
                animal = Parrot(name, age, weight)

            shelter.add_animal(animal)

        elif choice == "2":
            name = input("Имя животного: ")
            shelter.remove_animal(name)

        elif choice == "3":
            shelter.show_all()

        elif choice == "4":
            shelter.save_to_file()

        elif choice == "5":
            shelter.save_to_file()
            break


main()

4️⃣ app.py (Flask)

Python
from flask import Flask, render_template, request, redirect
from animals import Dog, Cat
from shelter import Shelter

app = Flask(__name__)
shelter = Shelter()

@app.route("/", methods=["GET", "POST"])
def home():
    if request.method == "POST":
        name = request.form["name"]
        type_animal = request.form["type"]
        if type_animal == "dog":
            animal = Dog(name, 2)
        else:
            animal = Cat(name, 2)
        shelter.add(animal)
        return redirect("/")

    return render_template("index.html", animals=shelter.get_all())

if __name__ == "__main__":
    app.run(debug=True)
